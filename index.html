$(document).ready(function() { var sliderImages = $('.slider-images'); var sliderImagesCount = sliderImages.children().length; var currentSlide = 1; var slideWidth = $('.slider').width(); var isAnimating = false; function moveToSlide(slide) { isAnimating = true; sliderImages.css('transform', 'translateX(-' + slideWidth * (slide - 1) + 'px)'); sliderImages.on('transitionend webkitTransitionEnd oTransitionEnd', function() { isAnimating = false; }); } function nextSlide() { if (!isAnimating) { if (currentSlide === sliderImagesCount) { currentSlide = 1; } else { currentSlide++; } moveToSlide(currentSlide); } } function prevSlide() { if (!isAnimating) { if (currentSlide === 1) { currentSlide = sliderImagesCount; } else { currentSlide--; } moveToSlide(currentSlide); } } var interval = setInterval(nextSlide, 5000); $('.slider').hover(function() { clearInterval(interval); }, function() { interval = setInterval(nextSlide, 5000); }); $('.slider-control.right').click(function() { if (!isAnimating) { nextSlide(); } }); $('.slider-control.left').click(function() { if (!isAnimating) { prevSlide(); } });
